		\knowncommand
	}

\colorlet{accB}{green!60!black}
\updatehighlight{
	name = accentB,
	color = accB,
	commands = {
		\inAccB
	},
	keywords = {
		$
	}
}


\colorlet{accC}{red!60!black}

\updatehighlight{
	name = accentC,
	color = accC,
	commands = {
		\inAccC	
	}
}

\colorlet{accD}{orange!100!black}

\updatehighlight{
	name = accentD,
	color = accD,
	commands = {
		\inAccD
	}
}

\begin{document}
	\begin{frame}
		\titlepage
		\centering
		Let's start
	\end{frame}

	\begin{saveblock}{simpledocument}
		\begin{highlightblock}[linewidth=19em,gobble=4]
			% File: document.tex
			\documentclass[a4paper]{article}
			\usepackage{amsmath}
			
			\begin{document}
				Insert math like $\sqrt{2}$.
				
			\end{document}
		\end{highlightblock}
	\end{saveblock}
	
	\begin{frame}
		\frametitle{Simple document}
		And your code should now look like:
		\consumeblock{simpledocument}
		after it
	\end{frame}
\end{document}
