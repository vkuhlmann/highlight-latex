\begin{document}
	\begin{multicols*}{2}
		For this, change the maximal number depth
		\begin{highlightblock}
			% Here is some code
			\setcounter{secnumdepth}{1}
			\begin{document}
				\section{With number (depth 1)}
				
				\setcounter{secnumdepth}{0}
				\section{No number}
				\unknowncommand\knowncommand
				\inAccB\inAccC\inAccD
				\section ~\smash{\ensuremath{\sqrt{2}\;\leftarrow}} sneaky!~
			\end{document}
		\end{highlightblock}
		and some text after it.
		
		Let's also show square brackets:
		\begin{highlightblock}[linewidth=19em]
			% File: document.tex
			\documentclass[a4paper]{article}
			\usepackage{amsmath}
			
			\begin{document}
				Insert math like $\sqrt{2}$.
				
			\end{document}
		\end{highlightblock}
	
		We can also have code inline, like \lstinline|\thesection|, or \lstinline|\author{you}|.
